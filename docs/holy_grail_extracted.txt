7.3 7.3 Kalman Filter for Dynamic Beta (Real-Time)
ThefollowingimplementstherecursiveKalmanfiltertoupdatetheequitybeta (βt)daily.
class KalmanBeta:
def __init__(self, beta0=1.0, sigma_beta=0.01, sigma_obs=0.02):
self.beta = beta0 # State estimate
self.P = 1.0 # Error covariance
self.Q = sigma_beta**2 # Process noise
self.R = sigma_obs**2 # Observation noise
def update(self, market_return, stock_return):
# Prediction step
beta_pred = self.beta
P_pred = self.P + self.Q
# Kalman Gain
K = P_pred / (P_pred + self.R)
# Correction step (using GBM observation equation)
residual = stock_return - (beta_pred * market_return)
self.beta = beta_pred + K * residual
self.P = (1 - K) * P_pred
return self.beta
